        function calculateROI() {

            const annualDischarges = parseFloat(document.getElementById('annualDischarges').value) || 0;

            const costPerReadmission = parseFloat(document.getElementById('costPerReadmission').value) || 0;

            const programCost = parseFloat(document.getElementById('programCost').value) || 0;

            const alos = parseFloat(document.getElementById('alos').value) || 0;

            const unreachedRatePct = parseFloat(document.getElementById('unreachedRate').value) / 100;

            const currentRatePct = parseFloat(document.getElementById('currentRate').value) / 100;


            document.getElementById('unreachedRateDisplay').innerText = (unreachedRatePct * 100).toFixed(1) + '%';

            document.getElementById('currentRateDisplay').innerText = (currentRatePct * 100).toFixed(1) + '%';

            

            const variancePct = currentRatePct - unreachedRatePct;

            const varElem = document.getElementById('rateVarianceDisplay');

            varElem.innerText = (variancePct * 100).toFixed(1) + '%';

            varElem.className = variancePct <= 0 ? "text-lg font-bold text-[#00B4A4]" : "text-lg font-bold text-red-500";


            // Calculations

            const baselineReadmissions = Math.round(annualDischarges * unreachedRatePct);

            const baselineTotalCost = baselineReadmissions * costPerReadmission;

            const baselineBedDays = Math.round(baselineReadmissions * alos);


            const actualReadmissions = Math.round(annualDischarges * currentRatePct);

            const actualTotalCost = actualReadmissions * costPerReadmission;

            const actualBedDays = Math.round(actualReadmissions * alos);


            const relativeReduction = unreachedRatePct > 0 ? (unreachedRatePct - currentRatePct) / unreachedRatePct : 0;

            const readmissionsPrevented = Math.max(0, baselineReadmissions - actualReadmissions);

            const bedDaysSaved = Math.max(0, baselineBedDays - actualBedDays);

            const grossSavings = Math.max(0, baselineTotalCost - actualTotalCost);

            const netSavings = grossSavings - programCost;

            const roiPercent = programCost > 0 ? (netSavings / programCost) * 100 : 0;

            

            calcState = {

                annualDischarges, costPerReadmission, programCost, alos,

                unreachedRatePct, currentRatePct, variancePct,

                baselineReadmissions, baselineTotalCost, baselineBedDays,

                actualReadmissions, actualTotalCost, actualBedDays,

                relativeReduction, readmissionsPrevented, bedDaysSaved, grossSavings, netSavings, roiPercent

            };


            // Update UI

            const fmt = new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', maximumFractionDigits: 0 });

            

            document.getElementById('grossSavings').innerText = fmt.format(grossSavings);

            document.getElementById('reductionLabel').innerText = Math.max(0, Math.round(relativeReduction * 100)) + '% total reduction';

            document.getElementById('preventedCount').innerText = readmissionsPrevented.toLocaleString();

            document.getElementById('bedDaysSaved').innerText = bedDaysSaved.toLocaleString();

            document.getElementById('netSavings').innerText = fmt.format(netSavings);

            

            const roiDisplay = document.getElementById('roiPercentDisplay');

            roiDisplay.innerText = `ROI: ${Math.round(roiPercent)}%`;

            roiDisplay.className = roiPercent >= 0 ? "text-sm text-white/60 mt-3 font-medium" : "text-sm text-red-300 mt-3 font-bold uppercase";


            costChart.data.datasets[0].data = [baselineTotalCost, actualTotalCost];

            costChart.update();

            volumeChart.data.datasets[0].data = [baselineReadmissions, actualReadmissions];

            volumeChart.update();


            document.getElementById('breakdownTable').innerHTML = `

                <tr>

                    <td class="px-6 py-4">Readmission Rate</td>

                    <td class="px-6 py-4 text-slate-500">${(unreachedRatePct*100).toFixed(1)}%</td>

                    <td class="px-6 py-4 text-[#5f0957]">${(currentRatePct*100).toFixed(1)}%</td>

                    <td class="px-6 py-4 text-[#00B4A4]">${(variancePct*100).toFixed(1)}%</td>

                </tr>

                <tr>

                    <td class="px-6 py-4">Readmission Volume</td>

                    <td class="px-6 py-4 text-slate-500">${baselineReadmissions.toLocaleString()}</td>

                    <td class="px-6 py-4 text-[#5f0957]">${actualReadmissions.toLocaleString()}</td>

                    <td class="px-6 py-4 text-[#00B4A4]">-${readmissionsPrevented.toLocaleString()} Patients</td>

                </tr>

                <tr>

                    <td class="px-6 py-4">Patient Days (Bed Days)</td>

                </tr>
